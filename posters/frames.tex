\define@key{portraittitleframe}{creator}[]{\def\ptf@creator{#1}}
\define@key{portraittitleframe}{inspector}[]{\def\ptf@inspector{#1}}
\define@key{portraittitleframe}{norminspector}[]{\def\ptf@norminspector{#1}}
\define@key{portraittitleframe}{approver}[]{\def\ptf@approver{#1}}
\define@key{portraittitleframe}{page}[]{\def\ptf@page{#1}}
\define@key{portraittitleframe}{total}[]{\def\ptf@total{#1}}
\define@key{portraittitleframe}{departmentandgroup}[]{\def\ptf@departmentandgroup{#1}}
\define@key{portraittitleframe}{tpga}[]{\def\ptf@tpga{#1}}
\define@key{portraittitleframe}{name}[]{\def\ptf@name{#1}}
\define@key{portraittitleframe}{paperformat}[]{\def\ptf@paperformat{#1}}
\define@key{portraittitleframe}{namestyle}[]{\def\ptf@namestyle{#1}}

% Рамка заглавной страницы бумажного формата
\newcommand{\portraittitleframe}[1][]{
\setkeys{portraittitleframe}{
	creator,
	inspector,
	norminspector,
	approver,
	page,
	total,
	departmentandgroup,
	tpga,
	name,
	paperformat,
	namestyle,
}
\setkeys{portraittitleframe}{#1}
\simpleframe
\titletable[
	tpga=\ptf@tpga,
	departmentandgroup=\ptf@departmentandgroup,
	name=\ptf@name,
	creator=\ptf@creator,
	inspector=\ptf@inspector,
	norminspector=\ptf@norminspector,
	approver=\ptf@approver,
	page=\ptf@page,
	total=\ptf@total,
	namestyle=\ptf@namestyle
]
\topinsсriptionforportrait[\ptf@tpga]
\paperformat{\ptf@paperformat}
}


\makeatother
