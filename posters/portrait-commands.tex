


\usepackage{ifthen}

\makeatletter

% Определение команды с ключами и значениями
\define@key{titleposter}{page}[]{\def\titleposter@page{#1}}
\define@key{titleposter}{total}[]{\def\titleposter@total{#1}}
\define@key{titleposter}{number}[]{\def\titleposter@number{#1}}
\define@key{titleposter}{name}[]{\def\titleposter@name{#1}}
\define@key{titleposter}{namestyle}[\LARGE]{\def\titleposter@namestyle{#1}}

\newcommand{\titletpga}{
	\ifthenelse
	{\equal{\titleposter@number}{}}
	{\tpga~\doctype}
	{\tpga-\titleposter@number~\doctype}}
% Рамка заглавной страницы бумажного формата
\newcommand{\titleposter}[2][]{
	\setkeys{titleposter}{
		page,
		total,
		number,
		name,
		namestyle
	}
	\setkeys{titleposter}{#1}
	\newpage

	\portraittitleframe[
		creator=\authorinframe,
		inspector=\inspectorinframe,
		norminspector=\norminspectorinframe,
		approver=\approverinframe,
		page=\titleposter@page,
		total=\titleposter@total,
		departmentandgroup=\departmentandgroupinframe,
		tpga=\titletpga,
		name=\titleposter@name,
		paperformat=A1,
		namestyle=\titleposter@namestyle
	]\HUGE#2
}


\makeatother

\makeatletter

\define@key{subsequentposter}{number}[]{\def\subsequentposter@number{#1}}
\newcommand{\subsequenttpga}{
	\ifthenelse
	{\equal{\subsequentposter@number}{}}
	{\tpga~\doctype}
	{\tpga-\subsequentposter@number~\doctype}}

% Рамка заглавной страницы бумажного формата
\newcommand{\subsequentposter}[2][]{
	\setkeys{subsequentposter}{
		page,
		number,
	}
	\setkeys{subsequentposter}{#1}

	\portraitsubsequentframe[
		tpga=\subsequenttpga,

		paperformat=A1,
	]\HUGE#2
}


\makeatother
